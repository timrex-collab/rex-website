Implementiere serverseitige OG-Tag-Injektion für Social-Media-Crawler 
via Netlify Edge Function. React-Helmet funktioniert für Crawler nicht 
(kein JS-Rendering) – die Edge Function injiziert statische Meta-Tags 
direkt in den <head> vor der Auslieferung.

---

function buildOGTags(og: OGData, url: string): string {
  return `
    <meta property="og:type" content="website" />
    <meta property="og:url" content="${url}" />
    <meta property="og:title" content="${og.title}" />
    <meta property="og:description" content="${og.description}" />
    <meta property="og:image" content="${og.image}" />
    ${og.imageWidth ? `<meta property="og:image:width" content="${og.imageWidth}" />` : ""}
    ${og.imageHeight ? `<meta property="og:image:height" content="${og.imageHeight}" />` : ""}
    <meta property="og:site_name" content="Rex Bedachungs GmbH" />
    <meta name="twitter:card" content="summary_large_image" />
    <meta name="twitter:title" content="${og.title}" />
    <meta name="twitter:description" content="${og.description}" />
    <meta name="twitter:image" content="${og.image}" />
  `.trim();
}

export default async function handler(request: Request, context: Context) {
  const url = new URL(request.url);
  const pathname = url.pathname.replace(/\/$/, "") || "/";

  // Nur HTML-Requests verarbeiten
  const accept = request.headers.get("accept") || "";
  if (!accept.includes("text/html")) {
    return context.next();
  }

  const response = await context.next();
  const contentType = response.headers.get("content-type") || "";
  if (!contentType.includes("text/html")) {
    return response;
  }

  const ogData = routes[pathname] ?? {
    title: "Rex Bedachungs GmbH Bochum",
    description: "Ihr Dachdecker in Bochum. Tel. 0234 583100.",
    image: `${BASE_URL}/images/dach-hintergrund-rex-bedachung.webp`,
    imageWidth: "1200",
    imageHeight: "630",
  };

  const ogTags = buildOGTags(ogData, `${BASE_URL}${pathname}`);
  const html = await response.text();

  // OG-Tags direkt nach <head> injizieren
  const modified = html.replace("<head>", `<head>\n${ogTags}`);

  return new Response(modified, {
    status: response.status,
    headers: response.headers,
  });
}

export const config = { path: "/*" };

## SCHRITT 2 – netlify.toml anpassen

Prüfe ob netlify.toml existiert. Falls ja, ergänze:

[[edge_functions]]
  path = "/*"
  function = "og-inject"

Falls netlify.toml noch nicht existiert, neu anlegen mit:

[build]
  command = "npm run build"
  publish = "dist/public"

[[edge_functions]]
  path = "/*"
  function = "og-inject"
